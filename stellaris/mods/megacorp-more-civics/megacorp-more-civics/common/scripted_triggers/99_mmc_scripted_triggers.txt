prosperity_mmc_militarist_check_trigger = {
	is_militarist = yes
	planet = {
		exists = owner
		OR = {
			owner = { has_valid_civic = civic_mmc_militarist }
			AND = {
				has_branch_office = yes
				branch_office_owner = { has_valid_civic = civic_mmc_militarist }
			}
		}	
	}
}
